type book_type: [AA, BB];

struct book{

   kind : book_type;

   when AA book {

      act() is {

         outf("AA: ");

      };

   };

   when BB book {

      act() is {

         outf("BB: ");

      };

   };

};

extend sys {

   book_list[20] : list of book;

   run is also {

      for each AA book (n) in book_list {

         out(index);

         n.act();

      }

      for each BB book (n) in book_list do {

         out(index);

         n.act();

      }

};
